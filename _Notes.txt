tar -xvzf wasi-sdk-30.0-x86_64-linux.tar.gz

// WASM Micro Runtime
https://github.com/bytecodealliance/wasm-micro-runtime/blob/main/product-mini/README.md


// export commands (Currently using second one)
export WASI_SDK_PATH=wasi-sdk-30.0-x86_64-linux
export WASI_SDK_PATH=wasi-sdk-20.0+threads
export WAMR_PATH=wasm-micro-runtime/product-mini/platforms/linux/build

// All in one line
export WASI_SDK_PATH=wasi-sdk-20.0+threads && export WAMR_PATH=wasm-micro-runtime/product-mini/platforms/linux/build

// basic compile command
$WASI_SDK_PATH/bin/clang --sysroot $WASI_SDK_PATH/share/wasi-sysroot -O2 example.c -o example.wasm

// main compile command
$WASI_SDK_PATH/bin/clang --sysroot $WASI_SDK_PATH/share/wasi-sysroot --target=wasm32-wasi-threads -pthread -Wl,--import-memory,--export-memory,--max-memory=67108864 main.c spinlock.c yieldSpinlock.c blockMutex.c -o main.wasm

// CV Demo compile
$WASI_SDK_PATH/bin/clang --sysroot $WASI_SDK_PATH/share/wasi-sysroot --target=wasm32-wasi-threads -pthread -Wl,--import-memory,--export-memory,--max-memory=67108864 cvDemo.c conditionVars.c blockMutex.c semaphore.c -o cv.wasm

//basic run command
$WAMR_PATH/iwasm --max-threads=15 main.wasm

$WAMR_PATH/iwasm --max-threads=100 cv.wasm
